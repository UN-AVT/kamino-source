Ingest the data

url_root <- "https://raw.githubusercontent.com/UN-AVT/kamino-source/main/sources/0-shared/data/"
url_file <- "global-slavery-index/data.csv"
url <- paste0(url_root, url_file)

df_slavery <- read.csv(url, header = TRUE, stringsAsFactors = FALSE)
df_slavery

Wrangle

Rename all columns using gsub() and regex

#Renaming column names
df1 <- df_slavery
colnames(df1) <- gsub("\\.|\\_", " ",colnames(df1))
df1

Another approach
Convert all columns using rename_with() and gsub()

df2 <- df_slavery

#rename_with() renames columns using a function
df2 <- df2 %>%  rename_with(~ gsub('[[:punct:]]',' ',.x))
df2
